-- Crear la tabla Vivienda
CREATE TABLE Vivienda (
    ViviendaID INT PRIMARY KEY,
    Direccion VARCHAR(255) NOT NULL,
    Ciudad VARCHAR(100) NOT NULL,
    Estado VARCHAR(100) NOT NULL,
    CodigoPostal VARCHAR(20) NOT NULL
);
 
-- Crear la tabla Empleado
CREATE TABLE Empleado (
    EmpleadoID INT PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL,
    Apellido VARCHAR(100) NOT NULL,
    FechaNacimiento DATE NOT NULL,
    ViviendaID INT,
    FOREIGN KEY (ViviendaID) REFERENCES Vivienda(ViviendaID)
);
 
-- Crear la tabla Oficio
CREATE TABLE Oficio (
    OficioID INT PRIMARY KEY,
    Descripcion VARCHAR(255) NOT NULL
);
 
-- Crear la tabla Oficio_Empleado
CREATE TABLE Oficio_Empleado (
    OficioEmpleadoID INT PRIMARY KEY,
    EmpleadoID INT,
    OficioID INT,
    FechaAsignacion DATE NOT NULL,
    FOREIGN KEY (EmpleadoID) REFERENCES Empleado(EmpleadoID),
    FOREIGN KEY (OficioID) REFERENCES Oficio(OficioID)
);
